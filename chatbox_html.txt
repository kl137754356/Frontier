    <script>
        // ======================
        // 元素引用
        // ======================
        const floatBall = document.getElementById('floatBall');
        const chatWindow = document.getElementById('chatWindow');
        const chatBody = document.getElementById('chatBody');
        const inputMsg = document.getElementById('inputMsg');
        const sendBtn = document.getElementById('sendBtn');
        const contextMenu = document.getElementById('contextMenu');

        // ======================
        // 悬浮球拖动 + 点击切换
        // ======================
        let isDragging = false;
        let hasMoved = false;
        let startX, startY, ballX, ballY;

        // 根据悬浮球位置定位聊天窗口
        function positionChatWindow() {
            const rect = floatBall.getBoundingClientRect();
            const winW = window.innerWidth;
            const winH = window.innerHeight;
            const chatW = 420;
            const chatH = 680;
            const gap = 12;

            let left, top;

            // 水平方向：优先放在悬浮球左侧，放不下则放右侧
            if (rect.left - chatW - gap >= 0) {
                left = rect.left - chatW - gap;
            } else if (rect.right + gap + chatW <= winW) {
                left = rect.right + gap;
            } else {
                left = Math.max(0, Math.min(rect.left - chatW - gap, winW - chatW));
            }

            // 垂直方向：底部对齐悬浮球底部，超出则调整
            top = rect.bottom - chatH;
            if (top < 0) top = 0;
            if (top + chatH > winH) top = winH - chatH;

            chatWindow.style.left = left + 'px';
            chatWindow.style.top = top + 'px';
        }

        floatBall.addEventListener('mousedown', (e) => {
            if (e.button !== 0) return; // 只响应左键
            isDragging = true;
            hasMoved = false;
            startX = e.clientX;
            startY = e.clientY;
            const rect = floatBall.getBoundingClientRect();
            ballX = rect.left;
            ballY = rect.top;
            e.preventDefault();
        });

        document.addEventListener('mousemove', (e) => {
            if (!isDragging) return;
            const dx = e.clientX - startX;
            const dy = e.clientY - startY;
            if (Math.abs(dx) > 3 || Math.abs(dy) > 3) {
                hasMoved = true;
            }
            floatBall.style.left = (ballX + dx) + 'px';
            floatBall.style.top = (ballY + dy) + 'px';
            floatBall.style.right = 'auto';
            floatBall.style.bottom = 'auto';
            // 拖动时同步更新聊天窗口位置
            if (chatWindow.style.display === 'flex') {
                positionChatWindow();
            }
        });

        document.addEventListener('mouseup', (e) => {
            if (!isDragging) return;
            isDragging = false;
            // 如果没有移动，视为点击 → 切换聊天窗口
            if (!hasMoved) {
                if (chatWindow.style.display === 'flex') {
                    chatWindow.style.display = 'none';
                } else {
                    positionChatWindow();
                    chatWindow.style.display = 'flex';
                }
            }
        });

        // ======================
        // 右键菜单
        // ======================
        floatBall.addEventListener('contextmenu', (e) => {
            e.preventDefault();
            contextMenu.style.left = e.clientX + 'px';
            contextMenu.style.top = e.clientY + 'px';
            contextMenu.style.display = 'block';
        });

        // 点击其他区域关闭菜单
        document.addEventListener('click', (e) => {
            if (!contextMenu.contains(e.target)) {
                contextMenu.style.display = 'none';
            }
        });

        // 菜单项点击
        contextMenu.addEventListener('click', (e) => {
            const item = e.target.closest('.menu-item');
            if (!item) return;
            const action = item.dataset.action;
            contextMenu.style.display = 'none';
            switch (action) {
                case 'login':
                    alert('登入功能开发中...');
                    break;
                case 'settings':
                    alert('设置功能开发中...');
                    break;
                case 'about':
                    alert('Polaris 助手 v1.0');
                    break;
            }
        });

        // ======================
        // 发送消息
        // ======================
        function sendMessage() {
            const msg = inputMsg.value.trim();
            if (!msg) return;

            // 用户消息
            chatBody.innerHTML += `<div class="message user-msg">${msg}</div>`;
            inputMsg.value = '';
            autoResizeTextarea();
            chatBody.scrollTop = chatBody.scrollHeight;

            // AI 回复（模拟）
            setTimeout(() => {
                const reply = `
这是一段可运行的代码：
<pre><code>function hello() {
  console.log("Hello Frontier!");
}</code></pre>
需要我帮你解释或修改吗？`;

                chatBody.innerHTML += `<div class="message ai-msg">${reply}</div>`;
                chatBody.scrollTop = chatBody.scrollHeight;
            }, 800);
        }

        sendBtn.addEventListener('click', sendMessage);
        inputMsg.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });

        // ======================
        // 输入框自动扩展高度
        // ======================
        function autoResizeTextarea() {
            inputMsg.style.height = 'auto';
            const maxH = 340 - 44; // 聊天窗口一半高度减去操作栏
            const scrollH = inputMsg.scrollHeight;
            if (scrollH > maxH) {
                inputMsg.style.height = maxH + 'px';
                inputMsg.style.overflowY = 'auto';
            } else {
                inputMsg.style.height = scrollH + 'px';
                inputMsg.style.overflowY = 'hidden';
            }
        }

        inputMsg.addEventListener('input', autoResizeTextarea);

        // ======================
        // 截图工具
        // ======================
        const screenshotBtn = document.getElementById('screenshotBtn');
        screenshotBtn.addEventListener('click', () => {
            alert('截图工具启动中...');
            // TODO: 接入实际截图功能
        });

        // ======================
        // 附件上传
        // ======================
        const attachBtn = document.getElementById('attachBtn');
        const fileInput = document.getElementById('fileInput');

        attachBtn.addEventListener('click', () => {
            fileInput.click();
        });

        fileInput.addEventListener('change', (e) => {
            const files = e.target.files;
            if (!files.length) return;
            for (const file of files) {
                chatBody.innerHTML += `<div class="message user-msg">📎 ${file.name}</div>`;
            }
            chatBody.scrollTop = chatBody.scrollHeight;
            fileInput.value = '';
        });

        // ======================
        // 新建对话 + 历史对话
        // ======================
        const newChatBtn = document.getElementById('newChatBtn');
        const historyBtn = document.getElementById('historyBtn');
        const historyPanel = document.getElementById('historyPanel');
        const closeHistoryBtn = document.getElementById('closeHistoryBtn');
        const historyList = document.getElementById('historyList');

        // 对话存储
        let conversations = [];
        let currentConvId = null;

        // 初始化第一个对话
        function initFirstConversation() {
            const conv = {
                id: Date.now(),
                title: '新对话',
                time: new Date().toLocaleString(),
                messages: chatBody.innerHTML
            };
            conversations.push(conv);
            currentConvId = conv.id;
        }
        initFirstConversation();

        // 新建对话
        newChatBtn.addEventListener('click', () => {
            // 保存当前对话
            saveCurrentConversation();
            // 创建新对话
            const conv = {
                id: Date.now(),
                title: '新对话',
                time: new Date().toLocaleString(),
                messages: '<div class="message ai-msg">你好！我是Hexagon助手，你可以问我各种问题～</div>'
            };
            conversations.push(conv);
            currentConvId = conv.id;
            chatBody.innerHTML = conv.messages;
        });

        // 保存当前对话内容
        function saveCurrentConversation() {
            const conv = conversations.find(c => c.id === currentConvId);
            if (conv) {
                conv.messages = chatBody.innerHTML;
                // 用第一条用户消息作为标题
                const firstUserMsg = chatBody.querySelector('.user-msg');
                if (firstUserMsg) {
                    conv.title = firstUserMsg.textContent.slice(0, 20) || '新对话';
                }
            }
        }

        // 打开历史面板
        historyBtn.addEventListener('click', () => {
            saveCurrentConversation();
            renderHistoryList();
            historyPanel.style.display = 'flex';
        });

        // 关闭历史面板
        closeHistoryBtn.addEventListener('click', () => {
            historyPanel.style.display = 'none';
        });

        // 渲染历史列表
        function renderHistoryList() {
            historyList.innerHTML = conversations.map(conv => `
                <div class="history-item ${conv.id === currentConvId ? 'active' : ''}" data-id="${conv.id}">
                    <div class="history-title">${conv.title}</div>
                    <div class="history-time">${conv.time}</div>
                </div>
            `).reverse().join('');
        }

        // 点击历史项切换对话
        historyList.addEventListener('click', (e) => {
            const item = e.target.closest('.history-item');
            if (!item) return;
            const id = Number(item.dataset.id);
            const conv = conversations.find(c => c.id === id);
            if (conv) {
                saveCurrentConversation();
                currentConvId = conv.id;
                chatBody.innerHTML = conv.messages;
                historyPanel.style.display = 'none';
            }
        });
    </script>
